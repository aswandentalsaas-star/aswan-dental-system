// ==========================================
// 1. Super Admin & Clinic Management
// ==========================================
enum PlanType {
  STARTER
  PROFESSIONAL
  ENTERPRISE
}

model Clinic {
  id              String           @id @default(cuid())
  name            String
  plan            PlanType         @default(STARTER)
  isActive        Boolean          @default(true)
  users           User[]
  patients        Patient[]
  appointments    Appointment[]
  inventory       InventoryItem[]
  suppliers       Supplier[]
  labs            Lab[]
  transactions    Transaction[]
  pettyCash       PettyCash[]
  createdAt       DateTime         @default(now())
}

enum Role {
  ADMIN
  DOCTOR
  RECEPTIONIST
  ACCOUNTANT
}

model User {
  id        String   @id @default(cuid())
  email     String   @unique
  password  String
  name      String
  role      Role     @default(DOCTOR)
  clinicId  String
  clinic    Clinic   @relation(fields: [clinicId], references: [id])
  createdAt DateTime @default(now())
}

// ==========================================
// 2 & 3. Patient Lifecycle & Appointments
// ==========================================
enum PatientStatus {
  MALE
  FEMALE
  CHILD
  PREGNANT
}

model Patient {
  id                String            @id @default(cuid())
  name              String
  age               String
  address           String
  phone             String
  emergencyPhone    String?           // هاتف طوارئ إضافي
  gender            PatientStatus     @default(MALE)
  chronicDiseases   String[]          // مصفوفة الأمراض المزمنة (لربطها بالألوان)
  consentFormLink   String?           // رابط الإقرار المرفوع سحابياً
  medicalReportLink String?           // رابط التقرير الطبي
  jawMappingData    Json?             // لتخزين حالة الفك التفاعلي (SVG)
  clinicId          String
  clinic            Clinic            @relation(fields: [clinicId], references: [id])
  appointments      Appointment[]
  treatmentRecords  TreatmentRecord[]
  xRays             XRay[]
  labWorks          LabWork[]
  createdAt         DateTime          @default(now())
  transactions     Transaction[]
}

model Appointment {
  id          String   @id @default(cuid())
  startTime   DateTime
  duration    Int      @default(30) // الحجز الافتراضي 30 دقيقة
  isEmergency Boolean  @default(false) // حالة الطوارئ (تسمح بـ 10 دقائق)
  status      String   @default("SCHEDULED") // SCHEDULED, DELAYED, COMPLETED, NO_SHOW
  patientId   String
  patient     Patient  @relation(fields: [patientId], references: [id])
  clinicId    String
  clinic      Clinic   @relation(fields: [clinicId], references: [id])
}

// ==========================================
// 4. Smart Stock & Supply Chain
// ==========================================
enum StockCategory {
  EQUIPMENT    // الرف الأول: أجهزة ومعدات
  CONSUMABLES  // الرف الثاني: خامات واستهلاكات
}

model Supplier {
  id        String          @id @default(cuid())
  name      String
  contact   String
  address   String?
  clinicId  String
  clinic    Clinic          @relation(fields: [clinicId], references: [id])
  items     InventoryItem[]
}

model InventoryItem {
  id            String        @id @default(cuid())
  name          String
  category      StockCategory @default(CONSUMABLES)
  quantity      Float
  unit          String        // قطعة، علبة، كرتونة
  minThreshold  Float         // الحد الأدنى للتنبيه الذكي (مثلا 1 للأجهزة، 3 أو 20% للخامات)
  clinicId      String
  clinic        Clinic        @relation(fields: [clinicId], references: [id])
  supplierId    String?
  supplier      Supplier?     @relation(fields: [supplierId], references: [id])
}

// ==========================================
// 5 & 6. Clinical, Labs & X-Ray
// ==========================================
model Lab {
  id        String    @id @default(cuid())
  name      String
  contact   String?
  clinicId  String
  clinic    Clinic    @relation(fields: [clinicId], references: [id])
  works     LabWork[]
}

model LabWork {
  id          String   @id @default(cuid())
  procedure   String   // نوع التركيبة المطلوبة
  status      String   @default("PENDING") // PENDING, SENT, RECEIVED
  sentDate    DateTime @default(now())
  receiveDate DateTime?
  cost        Float?
  labId       String
  lab         Lab      @relation(fields: [labId], references: [id])
  patientId   String
  patient     Patient  @relation(fields: [patientId], references: [id])
}

model XRay {
  id          String   @id @default(cuid())
  imageUrl    String   // رابط الصورة السحابي
  jawPosition String   // Upper, Lower, Full
  notes       String?  // تعليق الطبيب
  patientId   String
  patient     Patient  @relation(fields: [patientId], references: [id])
}

model TreatmentRecord {
  id          String   @id @default(cuid())
  patientId   String
  patient     Patient  @relation(fields: [patientId], references: [id])
  diagnosis   String
  procedure   String
  medications String?  // الأدوية الموصوفة (لربطها بـ Groq AI)
  cost         Float         @default(0)
  transactions     Transaction[]
  createdAt   DateTime @default(now())
}

// ==========================================
// 7. Finance & Debt Hub
// ==========================================
model Transaction {
  id              String           @id @default(cuid())
  amount          Float
  type            String           // INCOME (إيرادات) أو EXPENSE (مصروفات)
  description     String
  
  // الربط مع العيادة (موجود سابقاً)
  clinicId        String
  clinic          Clinic           @relation(fields: [clinicId], references: [id])
  
  // الربط مع المريض (إضافة جديدة)
  patientId       String?          
  patient         Patient?         @relation(fields: [patientId], references: [id])
  
  // الربط مع الزيارة/الإجراء الطبي (إضافة جديدة)
  treatmentId     String?          
  treatmentRecord TreatmentRecord? @relation(fields: [treatmentId], references: [id])

  createdAt       DateTime         @default(now())
}

model PettyCash {
  id          String   @id @default(cuid())
  amount      Float
  reason      String   // المصروفات النثرية
  clinicId    String
  clinic      Clinic   @relation(fields: [clinicId], references: [id])
  createdAt   DateTime @default(now())

export default async function AppointmentsPage() {
  // 1. تحديد نطاق "اليوم"
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  const tomorrow = new Date(today);
  tomorrow.setDate(tomorrow.getDate() + 1);

  // 2. جلب مواعيد اليوم فقط
  const appointments = await prisma.appointment.findMany({
    where: {
      startTime: { gte: today, lt: tomorrow }
    },
    include: { patient: true },
    orderBy: { startTime: 'asc' },
  });

  // 3. جلب قائمة المرضى للبحث (هذا الجزء ناقص في كودك الحالي)
  const patients = await prisma.patient.findMany({
    select: {
      id: true,
      name: true,
      phone: true,
    },
    orderBy: { name: 'asc' }
  });

  return (
    <div className="p-8 space-y-6" dir="rtl">
      {/* رأس الصفحة */}
      <div className="flex flex-col md:flex-row justify-between items-start md:items-center gap-4 mb-8">
        <div>
          <h1 className="text-3xl font-bold text-slate-800">أجندة مواعيد اليوم</h1>
          <p className="text-slate-500 mt-1">يوجد لديك {appointments.length} مواعيد مجدولة لهذا اليوم</p>
        </div>
        
        {/* استدعاء الـ Modal وتمرير المرضى له (بدلاً من الزر العادي) */}
        <AddAppointmentModal patients={patients} />
      </div>

      {/* قائمة المواعيد */}
<div className="grid gap-4">
  {appointments.length > 0 ? (
    appointments.map((apt) => (
      <div 
        key={apt.id} 
        className={`bg-white p-5 rounded-[2rem] border border-slate-100 shadow-sm transition-all flex flex-col md:flex-row md:items-center justify-between border-r-8 
          ${apt.status === 'COMPLETED' 
            ? 'grayscale opacity-60 border-r-slate-400 bg-slate-100' 
            : apt.isEmergency ? 'border-r-red-500 shadow-md' : 'border-r-blue-500'}`}
      >
        
        {/* 1. بيانات المريض */}
        <div className="flex items-center gap-4">
          <div>
            <h3 className="font-bold text-lg text-slate-800">{apt.patient?.name}</h3>
            <p className="text-slate-500 text-sm">{apt.patient?.phone}</p>
          </div>
        </div>

        {/* 2. الأزرار (إتمام الكشف + ملف المريض) */}
        <div className="flex gap-2 mt-4 md:mt-0">
          {apt.status !== 'COMPLETED' && (
            <CompleteButton id={apt.id} />
          )}
          
          <Link
            href={`/patients/${apt.patientId}`}
            className="px-6 py-3 bg-slate-900 text-white rounded-xl font-bold hover:bg-slate-800 transition-colors flex items-center justify-center gap-2"
          >
            ملف المريض <ChevronLeft size={18} />
          </Link>
        </div>
      </div>
    ))
  ) : (
    /* تم نقل التعليق للداخل لضمان سلامة الأقواس */
    <div className="text-center py-10 bg-white rounded-[2rem] border border-slate-100">
      <p className="text-slate-500 font-medium">لا توجد مواعيد أخرى اليوم</p>
    </div>
  )}
      </div>
    </div>
  );

import { completeAppointment } from "@/lib/actions";
import { CheckCircle } from "lucide-react";
import { useState } from "react";

export default function CompleteButton({ id }: { id: string }) {
  const [loading, setLoading] = useState(false);

  const handleComplete = async () => {
    if (!confirm("هل تريد تأكيد إتمام الكشف؟")) return;
    setLoading(true);
    await completeAppointment(id);
    setLoading(false);
  };

  return (
    <button 
      onClick={handleComplete}
      disabled={loading}
      className="px-4 py-3 bg-emerald-50 text-emerald-600 rounded-xl font-bold hover:bg-emerald-100 transition-colors flex items-center gap-2 disabled:opacity-50"
    >
      <CheckCircle size={18} />
      {loading ? "جاري..." : "إتمام الكشف"}
    </button>
  );
}
